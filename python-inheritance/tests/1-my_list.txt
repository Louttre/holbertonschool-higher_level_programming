#!/usr/bin/python3
import unittest
MyList = __import__('1-my_list').MyList

class TestMyList(unittest.TestCase):
    def test_empty_list(self):
        # Test case 1: Empty list
        empty_list = MyList()
        self.assertEqual(empty_list.print_sorted(), None)  # Sorting an empty list should return None

    def test_single_element_list(self):
        # Test case 2: Single element list
        single_element_list = MyList([42])
        self.assertEqual(single_element_list.print_sorted(), [42])  # Sorting a single-element list should return the same element

    def test_multiple_elements_list(self):
        # Test case 3: List with multiple elements
        multiple_elements_list = MyList([8, 6, 7, 5, 3, 0, 9])
        self.assertEqual(multiple_elements_list.print_sorted(), [0, 3, 5, 6, 7, 8, 9])  # Sorting a list with multiple elements should return sorted list
